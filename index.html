Check Whether a Number is Prime, Palindrome, or Armstrong

// description
Use conditional and loop statements to check: (a) Prime – divisible by only 1 and itself, (b) Palindrome – reads same forwards and backwards, (c) Armstrong – sum of cubes of digits equals the number.
// source code — program2.cpp
C++
#include<iostream>
#include<cmath>
using namespace std;

// Check Prime
bool isPrime(int n) {
    if (n <= 1) return false;
    for (int i = 2; i <= sqrt(n); i++)
        if (n % i == 0) return false;
    return true;
}

// Check Palindrome
bool isPalindrome(int n) {
    int orig = n, rev = 0;
    while (n != 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }
    return orig == rev;
}

// Check Armstrong (3-digit)
bool isArmstrong(int n) {
    int sum = 0, temp = n;
    while (temp) {
        int d = temp % 10;
        sum += d * d * d;
        temp /= 10;
    }
    return sum == n;
}

int main() {
    int num;
    cout << "Enter a number: "; cin >> num;

    cout << num << (isPrime(num)     ? " is Prime\n"     : " is NOT Prime\n");
    cout << num << (isPalindrome(num) ? " is Palindrome\n" : " is NOT Palindrome\n");
    cout << num << (isArmstrong(num)  ? " is Armstrong\n"  : " is NOT Armstrong\n");
    return 0;
}
// output (input: 153)
Enter a number: 153
153 is NOT Prime
153 is NOT Palindrome
153 is Armstrong

Passing Parameters to Functions by Value and by Reference

// description
Demonstrate pass-by-value (copy is passed; original unchanged) vs pass-by-reference (alias is passed; original can be modified). Use swap as a classic example.
// source code — program4.cpp
C++
#include<iostream>
using namespace std;

// Pass by Value — original unchanged
void swapByValue(int x, int y) {
    int temp = x;
    x = y;
    y = temp;
    cout << "Inside swapByValue: x=" << x << " y=" << y << endl;
}

// Pass by Reference — original modified
void swapByRef(int &x, int &y) {
    int temp = x;
    x = y;
    y = temp;
}

// Demonstrate with area calculation
void squareByValue(int n)   { n *= n; cout << "ByVal inside: " << n << endl; }
void squareByRef (int &n)  { n *= n; }

int main() {
    int a = 5, b = 10;

    cout << "Before: a=" << a << " b=" << b << endl;
    swapByValue(a, b);
    cout << "After swapByValue: a=" << a << " b=" << b << endl;

    swapByRef(a, b);
    cout << "After swapByRef:   a=" << a << " b=" << b << endl;

    int num = 4;
    squareByValue(num);
    cout << "After squareByValue: num=" << num << endl;
    squareByRef(num);
    cout << "After squareByRef:   num=" << num << endl;

    return 0;
}
// output
Before: a=5 b=10
Inside swapByValue: x=10 y=5
After swapByValue: a=5 b=10       ← original unchanged
After swapByRef:   a=10 b=5       ← original modified

Overloaded Functions to Calculate Area of Different Shapes

// description
Use function overloading — same function name area() with different parameter lists — to compute area of circle (1 arg), rectangle (2 args), and triangle (3 args / base-height).
// source code — program5.cpp
C++
#include<iostream>
#include<cmath>
using namespace std;

const double PI = 3.14159265;

// Overload 1: Circle — one argument (radius)
double area(double r) {
    return PI * r * r;
}

// Overload 2: Rectangle — two arguments (length, breadth)
double area(double l, double b) {
    return l * b;
}

// Overload 3: Triangle — three sides (Heron's formula)
double area(double a, double b, double c) {
    double s = (a + b + c) / 2.0;
    return sqrt(s * (s-a) * (s-b) * (s-c));
}

int main() {
    cout << fixed;
    cout.precision(4);

    cout << "Area of Circle    (r=5)        = "
         << area(5.0)            << endl;

    cout << "Area of Rectangle (4 x 6)      = "
         << area(4.0, 6.0)       << endl;

    cout << "Area of Triangle  (3, 4, 5)    = "
         << area(3.0, 4.0, 5.0) << endl;

    return 0;
}
// output
Area of Circle    (r=5)     = 78.5398
Area of Rectangle (4 x 6)  = 24.0000
Area of Triangle  (3, 4, 5) = 6.0000

#include<iostream>
using namespace std;

class Box {

    double length, width, height;

Overload Binary (+) and Unary (−) Operators for a Complex Class

// description
Implement a Complex number class and overload the binary + operator to add two complex numbers, and the unary - operator to negate a complex number.
// source code — program9.cpp
C++
#include<iostream>
using namespace std;

class Complex {
public:
    double real, imag;

    Complex(double r = 0, double i = 0) : real(r), imag(i) {}

    // Binary + operator overload
    Complex operator+(const Complex& c) const {
        return Complex(real + c.real, imag + c.imag);
    }

    // Unary - operator overload
    Complex operator-() const {
        return Complex(-real, -imag);
    }

    // Binary - operator overload (subtraction)
    Complex operator-(const Complex& c) const {
        return Complex(real - c.real, imag - c.imag);
    }

    void display(string label) const {
        cout << label << real
             << (imag >= 0 ? " + " : " - ")
             << abs(imag) << "i\n";
    }
};

int main() {
    Complex c1(3.0, 4.0), c2(1.5, -2.5);
    c1.display("c1 = ");
    c2.display("c2 = ");
    (c1 + c2).display("c1 + c2 = ");
    (c1 - c2).display("c1 - c2 = ");
    (-c1).display("  -c1  = ");
    return 0;
}
// output
c1     = 3 + 4i
c2     = 1.5 - 2.5i
c1+c2  = 4.5 + 1.5i
c1-c2  = 1.5 + 6.5i
-c1    = -3 - 4i

Runtime Polymorphism Using Virtual Functions

// description
Use virtual functions and a base class pointer to demonstrate runtime polymorphism (late binding). The correct overridden function is called based on the actual object type, not the pointer type.
// source code — program11.cpp
C++
#include<iostream>
using namespace std;

class Shape {
public:
    virtual void draw() {
        cout << "Drawing a generic Shape\n";
    }
    virtual double area() { return 0; }
    virtual ~Shape() {}  // Virtual destructor
};

class Circle : public Shape {
    double r;
public:
    Circle(double r) : r(r) {}
    void draw()    override { cout << "Drawing Circle\n"; }
    double area() override { return 3.14159 * r * r; }
};

class Rectangle : public Shape {
    double l, b;
public:
    Rectangle(double l, double b) : l(l), b(b) {}
    void draw()    override { cout << "Drawing Rectangle\n"; }
    double area() override { return l * b; }
};

int main() {
    Shape *ptr;

    Circle    c(7.0);
    Rectangle rect(4.0, 5.0);

    ptr = &c;    ptr->draw();
    cout << "Area = " << ptr->area() << endl;

    ptr = &rect; ptr->draw();
    cout << "Area = " << ptr->area() << endl;

    return 0;
}
// output
Drawing Circle
Area = 153.938
Drawing Rectangle
Area = 20

Function Template to Sort an Array & Class Template for a Stack

// description
Create a generic sortArray() function template (works on any data type) using Bubble Sort. Also implement a generic Stack<T> class template supporting push, pop, and display.
// source code — program12.cpp
C++
#include<iostream>
using namespace std;

// ── Function Template: Bubble Sort ─────────────
template<typename T>
void sortArray(T arr[], int n) {
    for (int i = 0; i < n-1; i++)
        for (int j = 0; j < n-i-1; j++)
            if (arr[j] > arr[j+1]) swap(arr[j], arr[j+1]);
}

// ── Class Template: Stack ───────────────────────
template<typename T>
class Stack {
    T data[100];  int top;
public:
    Stack() : top(-1) {}
    void push(T val) {
        if (top < 99) data[++top] = val;
        else cout << "Stack Overflow!\n";
    }
    void pop() {
        if (top >= 0) cout << "Popped: " << data[top--] << endl;
        else cout << "Stack Underflow!\n";
    }
    void display() {
        for (int i = top; i >= 0; i--) cout << data[i] << " ";
        cout << endl;
    }
};

int main() {
    int arr[] = {5,2,8,1,9};
    sortArray(arr, 5);
    cout << "Sorted int array: ";
    for (int x : arr) cout << x << " "; cout << endl;

    Stack<int> s;
    s.push(10); s.push(20); s.push(30);
    cout << "Stack: "; s.display();
    s.pop();
    cout << "After pop: "; s.display();
    return 0;
}
// output
Sorted int array: 1 2 5 8 9
Stack: 30 20 10
Popped: 30
After pop: 20 10

Exception Handling: Division by Zero & Invalid Input

// description
Use try, catch, and throw to handle: (a) division by zero exception, (b) invalid (negative) input exception, and (c) a custom exception class for domain-specific errors.
// source code — program14.cpp
C++
#include<iostream>
#include<stdexcept>
#include<string>
using namespace std;

// Custom Exception Class
class NegativeInputException {
    string msg;
public:
    NegativeInputException(string m) : msg(m) {}
    string what() { return msg; }
};

double safeDivide(double a, double b) {
    if (b == 0) throw runtime_error("Division by zero!");
    return a / b;
}

double squareRoot(double n) {
    if (n < 0)
        throw NegativeInputException("Cannot take sqrt of negative number!");
    double result = 1, x = n;
    for(int i=0;i<50;i++) result=(result+x/result)/2;
    return result;
}

int main() {
    try {
        cout << "10 / 2 = " << safeDivide(10,2) << endl;
        cout << "10 / 0 = " << safeDivide(10,0) << endl;
    }
    catch (runtime_error& e) {
        cout << "Caught: " << e.what() << endl;
    }

    try {
        cout << "sqrt(25) = " << squareRoot(25) << endl;
        cout << "sqrt(-4) = " << squareRoot(-4) << endl;
    }
    catch (NegativeInputException& e) {
        cout << "Caught: " << e.what() << endl;
    }
    return 0;
}
// output
10 / 2 = 5
Caught: Division by zero!
sqrt(25) = 5
Caught: Cannot take sqrt of negative number!

Read and Write Student Records Using File Operations

// description
Use C++ file streams (ofstream, ifstream) to write student records to a binary/text file and read them back. Demonstrate sequential file I/O for persistent data storage.
// source code — program15.cpp
C++
#include<iostream>
#include<fstream>
#include<string>
using namespace std;

struct Student {
    char  name[50];
    int   roll;
    float marks;
};

void writeRecords() {
    ofstream fout("students.dat", ios::binary);
    Student students[] = {
        {"Jitesh",  101, 92.5f},
        {"Rahul",   102, 88.0f},
        {"Priya",   103, 95.0f},
        {"Vikash",  104, 79.5f}
    };
    for (auto& s : students)
        fout.write((char*)&s, sizeof(Student));
    fout.close();
    cout << "4 records written to students.dat\n\n";
}

void readRecords() {
    ifstream fin("students.dat", ios::binary);
    if (!fin) { cout << "File not found!\n"; return; }
    Student s;
    cout << "Roll\tName\t\tMarks\n";
    cout << "----\t----\t\t-----\n";
    while (fin.read((char*)&s, sizeof(Student)))
        cout << s.roll << "\t" << s.name
             << "\t\t" << s.marks << endl;
    fin.close();
}

int main() {
    writeRecords();
    readRecords();
    return 0;
}
// output
4 records written to students.dat

Roll    Name        Marks
----    ----        -----
101     Jitesh      92.5
102     Rahul       88
103     Priya       95
104     Vikash      79.5
